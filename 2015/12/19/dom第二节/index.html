function getAt(ele,aa){
        var list = ele.attributes;
        if(!list){
            return null;
        }
        for(var i = 0;i<list.length;i++){
            if(list[i].nodeName==aa){
                return list[i].nodeValue;
            }
        }
        return null;
}
